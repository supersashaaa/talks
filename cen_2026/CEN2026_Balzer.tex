\section{Introduction}\label{introduction}

\begin{frame}{Introduction \& Previous Work}
\protect\phantomsection\label{introduction-previous-work}
\pause

\begin{itemize}
\tightlist
\item
  Optimization of dose finding and selection is a key goal in oncology
  drug research.
\item
  OPTIMUS Project: FDA initiative to improve dose optimization/dose
  finding.
\item
  Key challenge: Accurate estimation of therapeutic effects

  \begin{itemize}
  \tightlist
  \item
    True treatment effect is often overestimated in Phase II
  \item
    Phase III results may not confirm Phase II findings.
  \end{itemize}
\item
  Bias-correction methods were presented for normally distributed
  outcomes (e.g.~boostrap methods) (\citep{tianyu2024})
\item
  No existing methods for binary endpoints identified in a literature
  research.
\end{itemize}
\end{frame}

\begin{frame}{Setup}
\protect\phantomsection\label{setup}
\begin{itemize}
\item
  \(I\) active treatment arms with \(n\) patients.
\item
  Response \(X_{ij}\) for patient \(j\) in group \(i\) with
  \(X_{ij} \sim \textrm{Ber}(\theta_i)\) and \(i = 1, \dots, I\).
\item
  \(\mathbf{X_i} = (X_{i1}, \dots, X_{in})\) vector of responses for
  group \(i\) and \(\mathbf{X} = (\mathbf{X_1}, \dots, \mathbf{X_I}).\)
\item
  Parameter of interest is
  \(\theta_{max} = \max(\theta_1, \dots, \theta_I).\)
\item
  Estimator:
  \(\hat{\theta}(\mathbf{X}) = \max[\bar{\mathbf{X}}_1, \dots ,\bar{\mathbf{X}}_I]\).
\item
  Even if \(\bar{\mathbf{X}}_i = \mathrm{mean}(X_{i1}, \dots, X_{in})\)
  is an unbiased estimator, \(\hat{\theta}(\mathbf{X})\) is biased.
\end{itemize}
\end{frame}

\section{Simulation Study}\label{simulation-study}

\begin{frame}{Part I - Optimization of Bootstrap Repetitions}
\protect\phantomsection\label{part-i---optimization-of-bootstrap-repetitions}
\textbf{Aim}: Optimize the number of bootstrap repetitions \(B\) for the
single and double (non-)parametric bootstrap in binary endpoints, based
on bias and MSE.

\begin{itemize}
\tightlist
\item
  The number of bootstrap repetitions \(B\) is set to 80, 100, 200, 500,
  and 1000.
\end{itemize}
\end{frame}

\begin{frame}{Part II - Bayesian Hierarchical Models}
\protect\phantomsection\label{part-ii---bayesian-hierarchical-models}
\textbf{Aim}: Identify the optimal hyperprior (with corresponding
parameters) for a Bayesian hierarchical model to minimize bias and MSE.

The model is specified as:
\(Y_i \sim \mathrm{Bin}(n, \theta_i), \quad i = 1, \dots, 5\) with

\[ \mathrm{logit}(\theta_i) \sim \mathcal{N}(\mu, \sigma^2), \quad \mu \sim \mathcal{N}(-1.38, 100).\]

\begin{itemize}
\item
  Hyperpriors determine how much information is pooled across groups
  (i.e., \(\sigma^2\) controls degree of shrinkage).
\item
  The following MCMC settings are applied fo all fits: burn-in 2.000,
  number of chains 3, total MCMC iterations per chain: 10.000.
\end{itemize}
\end{frame}

\begin{frame}{Part II - Priors}
\protect\phantomsection\label{part-ii---priors}
The priors and corresponding parameters to evaluate are based on
\citep{Cunanan}:

\begin{itemize}
\tightlist
\item
  \(\mathrm{Unif}[0, a]\) for \(\log(\sigma), \sigma, \sigma^2\) with

  \begin{itemize}
  \tightlist
  \item
    \(a \in \{1, \dots,  1000\}\).
  \end{itemize}
\item
  Half-t for \(\sigma\) with

  \begin{itemize}
  \tightlist
  \item
    scale parameter \(s \in \{0.5, \dots,  1000\}\) and
  \item
    degrees of freedom \(df \in \{1, \dots,  500\}\).
  \end{itemize}
\item
  Gamma distribution \(\Gamma(a,b)\) for \(\tau = \frac{1}{\sigma^2}\)
  with

  \begin{itemize}
  \tightlist
  \item
    \(a \in \{0.005, \dots,  2 \}\), and
  \item
    \(b \in  \{0.0005, \dots, 50\}\).
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{General Setup}
\protect\phantomsection\label{general-setup}
\begin{columns}[T]
\begin{column}{0.48\linewidth}
The simulation study consists of two parts (10.000 repetitions each),
with the following settings:

\begin{itemize}
\tightlist
\item
  9 dose response relationships; 3 or \textbf{5} doses
  \citep{Ivanova2012}
\item
  Maximal response: 60\% or \textbf{80\%}
\item
  Sample size: n = 20 or \textbf{40} patients per dose level
\end{itemize}
\end{column}

\begin{itemize}
\tightlist
\item
  No relevant differences in bias or MSE across bootstrap iterations.
\item
  \(B = 80\) bootstrap iterations provide adequate bias correction.
\item
  Other simulation scenarios reached the same conclusion.
\end{itemize}

\begin{frame}{Part II: Summary}
\protect\phantomsection\label{part-ii-summary}
\begin{tabular}{l r r l r r }
\textbf{Method} & \multicolumn{4}{l}{\textbf{Parameter}} & \textbf{Bias (MSE)} \\ \hline
Uniform $\log(\sigma)$ & $a = $ & 2& &  & -0.0070 (0.00351)  \\
Uniform $\sigma$ & $a = $ & 100 & &  & -0.0047 (0.00371)  \\
Uniform $\sigma^2$ & $a = $ & 1000 & &  & 0.0003 (0.00357)  \\
Half-t & $df = $ & 100& s = & 500 & -0.0043 (0.00369)  \\
Gamma & $a = $ & 0.25& b = & 5 & 0.0004 (0.00335) 
\end{tabular}

\begin{itemize}
\tightlist
\item
  Uniform prior on \(\sigma^2\) with \(a = 1000\) and Gamma prior with
  \(a = 0.25\) and \(b = 5\) show the smallest bias.
\item
  MSE is very similar for most parameter values and hyperpriors, except
  of extreme outliers.
\item
  Prior choice depends on the true underlying dose-response scenario.
\end{itemize}
\end{frame}

\section{Conclusion}\label{conclusion}

\begin{frame}{Conclusion}
\protect\phantomsection\label{conclusion-1}
\begin{itemize}
\tightlist
\item
  \(B = 80\) bootstrap iterations are sufficient.
\item
  Bayesian methods show promising results in reducing bias.
\item
  Further scenarios for Bayesian methods and further methods will be
  investigated.
\end{itemize}
\end{frame}
